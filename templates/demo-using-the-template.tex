\hsgsection{Setup}

\begin{frame}[t,fragile]
  \frametitle{Three commands to a finished deck}

  \begin{enumerate}
    \setlength\itemsep{0.4em}
    \item Bootstrap a new deck folder:
      \begin{lstlisting}
bin/new-deck.sh ~/path/to/my-talk
      \end{lstlisting}
    \item Edit \texttt{deck.tex}: replace every \texttt{TODO}.
    \item Compile (3 passes for remember-picture overlays to settle):
      \begin{lstlisting}
pdflatex deck.tex
pdflatex deck.tex
pdflatex deck.tex
      \end{lstlisting}
  \end{enumerate}

  Or one shot with \texttt{latexmk deck.tex} (uses the bundled
  \texttt{latexmkrc}).
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Folder layout produced by \texttt{new-deck.sh}}

  \begin{lstlisting}
my-talk/
  deck.tex                  <- you edit this
  beamerthemeHSG.sty        <- theme (do not edit)
  latexmkrc                 <- compile config
  assets/                   <- every image asset
    hsg-logo-en.pdf, hsg-badge.png, ...
  fonts/                    <- vendored Gillius ADF (xelatex/lualatex)
    GilliusADF-Regular.otf, ...
  \end{lstlisting}

  The theme adds \texttt{\textbackslash graphicspath\{\{./assets/\}\{./\}\}}
  so \texttt{\textbackslash includegraphics\{hsg-logo-en.pdf\}} resolves
  whether assets are in \texttt{./assets/} or flat next to the deck.
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Preamble: set once}

  \begin{lstlisting}
\documentclass[aspectratio=169]{beamer}

  {Prof. Dr. Author Name}
  {author@unisg.ch}

  \end{lstlisting}
\end{frame}

\hsgsection{Slide types}

\begin{frame}[t]
  \frametitle{The cover slide \emph{(Titel Slide 3)}}

  \codebox{\textbackslash begin\{frame\}[plain]\textbackslash titlepage\textbackslash end\{frame\}}

  \vspace{0.6em}
  Just one line in the body. The cover photo comes from
  \texttt{\textbackslash coverimage\{...\}} in the preamble. The HSG
  green badge is layered behind the photo (the badge corners peek
  through where the photo's clip cuts off).

  \vspace{0.6em}
  \fcolorbox{HSGmuted!40}{white}{%
    \parbox{0.96\linewidth}{\scriptsize
      \textbf{Tip:} the cover photo's right edge is clipped with a
      slanted polygon. Pick a photo whose subject sits in the
      \emph{left} third of the frame so the clip does not crop it.}}
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Agenda \emph{(Agenda layout)}}

  \begin{lstlisting}
\hsgagenda{
  \item Introduction
  \item Methodology
  \item Results
  \item Discussion
  \item Conclusion
}
  \end{lstlisting}

  \vspace{0.4em}
  Use 4--6 items, 1--2 words each. The same items should appear in
  \texttt{\textbackslash hsgsetsections} so the sidebar progress matches.
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Section advance \emph{(silent, no slide rendered)}}

  \begin{lstlisting}
\hsgsection{Methods}     % advances the sidebar
% next slide opens "Methods" with a green sidebar marker
  \end{lstlisting}

  \vspace{0.4em}
  \texttt{\textbackslash hsgsection} does NOT render a divider slide.
  It only updates the left-margin progress strip. Place it
  immediately before the first content slide of each section.
\end{frame}

\begin{frame}[t]
  \frametitle{Standard content slide \emph{(Titel und Content)}}

  \codebox{%
    \textbackslash begin\{frame\}\\
    \quad\textbackslash frametitle\{Results\}\\
    \quad\textbackslash framesubtitle\{Detection performance\}\\
    \quad\textbackslash begin\{itemize\}\\
    \quad\quad\textbackslash item \textbackslash hsgemph\{AUC = 0.89\} on the held-out split\\
    \quad\quad\textbackslash item False-positive rate held at 2\,\%\\
    \quad\textbackslash end\{itemize\}\\
    \textbackslash end\{frame\}}

  \vspace{0.5em}
  \texttt{\textbackslash framesubtitle} renders as a tiny grey
  breadcrumb above the title (keep it under 6 words).
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Two-column comparison \emph{(Titel zwei Content)}}

  No HSG-specific helper; use Beamer's native \texttt{columns}.

  \begin{lstlisting}
\begin{columns}[T]
  \column{0.5\textwidth}
    \textbf{Symmetric}
    \begin{itemize}\item Shared key\item Fast\end{itemize}
  \column{0.5\textwidth}
    \textbf{Asymmetric}
    \begin{itemize}\item Key pair\item Slow\end{itemize}
\end{columns}
  \end{lstlisting}

  \vspace{0.4em}
  \textbf{Rule:} do NOT colour column headers HSGgreen. Reserve green
  for inline \texttt{\textbackslash hsgemph}.
\end{frame}

\begin{frame}[t]

\begin{frame}[t,fragile]
  \frametitle{Three-column layout}

  \begin{lstlisting}
\begin{columns}[T]
  \column{0.33\textwidth} ...
  \column{0.33\textwidth} ...
  \column{0.33\textwidth} ...
\end{columns}
  \end{lstlisting}

  \vspace{0.4em}
  Asymmetric splits work too: \texttt{0.6\textbackslash textwidth} +
  \texttt{0.4\textbackslash textwidth} for a wide-text + narrow-figure
  pattern.
\end{frame}

\begin{frame}[t]
  \frametitle{Three-column rendered}

  \begin{columns}[T]
    \column{0.32\textwidth}
      \textbf{Confidentiality}
      \begin{itemize}\setlength\itemsep{0pt}
        \item Encryption hides content
        \item Adversary sees only ciphertext
      \end{itemize}
    \column{0.32\textwidth}
      \textbf{Integrity}
      \begin{itemize}\setlength\itemsep{0pt}
        \item MAC detects tampering
        \item Hash chains for logs
      \end{itemize}
    \column{0.32\textwidth}
      \textbf{Availability}
      \begin{itemize}\setlength\itemsep{0pt}

        \item Rate-limiting against DoS
      \end{itemize}
  \end{columns}

  \vspace{0.6em}
  Three balanced columns with \texttt{[T]} top-alignment. The CIA triad
  fits naturally into this layout.
\end{frame}

\hsgsection{Workflows}

\begin{frame}[t,fragile]
  \frametitle{CLI agent workflow}

  An agent (Claude Code, Codex, Cursor, Aider) scaffolds a deck cold:

  \begin{lstlisting}
git clone https://github.com/brunobastosrodrigues/beamer-hsg.git
cd beamer-hsg
bin/new-deck.sh ~/talks/2026-conference
cd ~/talks/2026-conference
# fill in TODO markers in deck.tex
pdflatex deck.tex && pdflatex deck.tex && pdflatex deck.tex
  \end{lstlisting}

  \vspace{0.4em}
  Three files the agent reads first:
  \begin{itemize}
    \setlength\itemsep{0.2em}
    \item \texttt{templates/skeleton.tex} -- canonical starter
    \item \texttt{helpers.json} -- machine-readable API manifest
    \item \texttt{AGENTS.md} -- decision table + golden rules
  \end{itemize}
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{What \texttt{helpers.json} gives an agent}

  \begin{lstlisting}
{
  "frame_helpers": [
    {"name": "\\hsgstatement",
     "args": ["text"],
     "layout": "Statement 1",
     "purpose": "One bold green sentence at 36 pt."},
    ...
  ],
  "rules_to_follow": [
    "Three pdflatex passes...",
    "HSG green is reserved for...",
    ...
  ]
}
  \end{lstlisting}

  Every public macro: name, arity, master-layout, one-line purpose.
\end{frame}

\begin{frame}[t,fragile]
  \frametitle{Overleaf workflow}

  Repo is private, so use the release-ZIP route (no paid Overleaf
  plan needed):

  \begin{enumerate}
    \setlength\itemsep{0.2em}
    \item GitHub repo $\rightarrow$ \emph{Releases} $\rightarrow$
          download \emph{Source code (zip)} for the latest tag.
    \item Overleaf $\rightarrow$ \emph{New Project $\rightarrow$ Upload
          Project} $\rightarrow$ drop the ZIP.
    \item \emph{Project Settings $\rightarrow$ Compiler}:
          \begin{itemize}\setlength\itemsep{0pt}
            \item \textbf{XeLaTeX} for the Gill-Sans-Nova look
                  (vendored Gillius ADF picked up automatically).
            \item \textbf{pdfLaTeX} for the Lato fallback.
          \end{itemize}
    \item Open \texttt{templates/skeleton.tex}, set as
          \emph{Main document}, replace TODOs, recompile.
  \end{enumerate}

  Full instructions: \texttt{OVERLEAF.md} in the repo root.
\end{frame}

\hsgsection{Tips}

\begin{frame}[t]
  \frametitle{Golden rules}

  \begin{itemize}
    \setlength\itemsep{0.45em}
    \item \hsgemph{Never use HSG green for titles or headings.}
          Reserved for the sidebar, the page badge, and
          \texttt{\textbackslash hsgemph\{...\}}.
    \item \hsgemph{Three passes} of \texttt{pdflatex}. Two is not
          enough; remember-picture overlays drift on pass two.
    \item \hsgemph{Copy ALL assets} (or use \texttt{bin/new-deck.sh},
          which does it for you).
    \item \hsgemph{Keep slides spare.} If you find yourself
          \texttt{shrink}ing to fit, split the slide instead.
    \item \hsgemph{Cover and closing} are hand-tuned. Do not redefine
          their templates.
  \end{itemize}
\end{frame}

\begin{frame}[t]
  \frametitle{Common errors}

  \begin{tabular}{@{}p{4.4cm}p{8.6cm}@{}}
    \toprule
    \textbf{Symptom} & \textbf{Where to look} \\
    \midrule
    Title chrome misplaced &
      Compile two more times -- remember-picture needs three passes. \\[3pt]
    Logo or photo missing &
      An asset under \texttt{assets/} is missing. \\[3pt]
    Bullets render green &
      You added a \texttt{\textbackslash setbeamercolor} in your
      preamble. Remove it. \\[3pt]
    Sidebar progress empty &
      You did not call \texttt{\textbackslash hsgsetsections} in the
      preamble. \\[3pt]
    Page number clipped &
      4-digit numbers (1000+) do not fit. Split the deck. \\
    \bottomrule
  \end{tabular}
\end{frame}
